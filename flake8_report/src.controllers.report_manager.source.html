42 def load_data(self):
43 """
44 Charge les données des tournois et des joueurs depuis le fichier JSON.
45 """
46 if self.tournament_manager.filepath.exists():
47 with self.tournament_manager.filepath.open('r', encoding='utf-8') as file:
48 data = json.load(file)
49 self.tournament_manager.load_tournaments(data.get('tournaments', []))
50 players = []
51 for tournament in self.tournament_manager.tournaments:
52 players.extend(tournament.players)
53 self.user_manager.load_players(players)
54 else:
55 print("Starting with an empty dataset.")

128 def ask_to_export_report(self, filename, display_function, data):
129 """
130 Demande à l'utilisateur s'il souhaite exporter le rapport et dans quel format.
131  
132 Args:
133 filename (str): Le nom de base du fichier de rapport.
134 display_function (function): La fonction utilisée pour générer le contenu du rapport.
135 data (any): Les données à passer à la fonction de génération de rapport.
136 """
137 choice = input("Souhaitez-vous imprimer ce rapport ? (Oui/Non) : ")
138 if choice.lower() == 'oui':
139 format_choice = input("Choisissez le format : 1. Texte brut, 2. HTML, 3. Les deux : ")
140 if format_choice == '1':
141 self.export_manager.export_report(filename + ".txt", display_function, data)
142 elif format_choice == '2':
143 self.export_manager.export_report_html(filename + ".html", display_function, data)
144 elif format_choice == '3':
145 self.export_manager.export_report(filename + ".txt", display_function, data)
146 self.export_manager.export_report_html(filename + ".html", display_function, data)
